js中几种for循环的对比

基本for循环

基本for循环在使用上没有其他限制，但是代码上写起来复杂一点儿

var array = [1,2,3];
for(var i = 0,len=array.length; i < len; i++) {
console.log(array[i]);//元素值
console.log(i);//元素对应的索引(下标)
}

for in

for in 一般会用于遍历对象，会深度遍历对象的每个键，即如果一个键对应的键值是对象obj的话，for in还会去遍历一遍这个对象obj。不建议遍历数组

var obj ={
a:1,
b:2,
c:{
a:1,
b:2
}
}
for (var i in obj) {
console.log(i);//key值
console.log(obj[i]);//key对应的value
}

forEach

forEach遍历数组很方便，但是缺点是不能执行break

var array = [1,2,3];
array.forEach(function(value,index){
console.log(value);//元素值
console.log(index);//元素对应的索引(下标)
});

for of

for of 是es6中新加的api，既具备了for in 和forEach在语法上的简练，又避免的深度遍历和不能break的缺点，遍历数组是很好的选择

let array = [1,2,3];
for (const i of array) {
console.log(i);//元素值
console.log(array.indexOf(i));//元素对应的索引(下标)
}
